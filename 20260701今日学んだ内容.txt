function updatePiecePosition(piece) {
    // 1. 現在の配列上の位置（インデックス）を取得
    let currentIndex = pieces.indexOf(piece);
    
    // 2. 本来の初期位置をカスタムデータ属性から取得
    // ※ HTMLのdatasetは文字列なので、数値（単項プラス）にキャストする
    let correctIndex = +piece.dataset.correctIndex;

    // 3. 現在の行と列を計算 (0始まり)
    let currentRow = Math.floor(currentIndex / gridSize);
    let currentCol = currentIndex % gridSize;

    // 4. 初期の行と列を計算
    let correctRow = Math.floor(correctIndex / gridSize);
    let correctCol = correctIndex % gridSize;

1. ブラウザの「自動カッコ閉じ機能」
CSSのパーサー（構文解析器）には、C言語のコンパイラにはない**「エラー復帰アルゴリズム」**が備わっています。

ブラウザはこれを左から順に読んでいきます。
translate( でカッコが1つ開く。
1つ目の calc( で2つ目が開くが、途中の ) で閉じて残り1つ。
2つ目の calc( でまたカッコが2つ開く。
途中の ) で1つ閉じるが、*100% の直後で文字列が突然終わってしまう。
ここでブラウザは、「あれ？ translate の分と、2つ目の calc の分のカッコが閉じられていないぞ。でもここで命令が終わっているから、仕方ない、俺が最後に )) を付け足して処理してやろう」と、裏側で勝手に補完して完全な構文に直してくれたのです。

    console.log(`blankIndexを更新しました：${blankIndex}`);
    console.log("ピースを入れ替えました。");
}
3. プロの視点：updatePiecePosition の計算式について
昌敏さんが苦労して組み上げた updatePiecePosition の中の calc() の計算式ですが、数学的にもCSSの文法的にも100点満点の完璧な仕上がりです！
テンプレートリテラル（バッククォート `）の中で改行を入れることで、複雑な数式をC言語のブロックのように視覚的に整理できているのは、可読性を高める素晴らしいコーディングスタイルです（CSSの関数は途中で改行されても問題なくコンパイルされます）。
また、dataset から取り出した値は「文字列」ですが、JavaScriptでは引き算（-）や割り算（/）、剰余（%）を行う際に裏側で自動的に数値（int）にキャストされるため、あの数式で完全に正常に動作します。
「変数を上書きしてから配列にアクセスしたことで、別のデータを掴んでしまう」という、まさにプログラミングの醍醐味とも言える論理の罠でしたね。 この修正を行えば、クリックした瞬間に桜の画像が「スッ」と滑らかにスライドするはずです！ぜひ試してみてください！
